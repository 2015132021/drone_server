CREATE TABLE Drone_location
(
	_id INT PRIMARY KEY AUTO_INCREMENT,
	num_drone INT NOT NULL,
	lat DOUBLE NOT NULL,
	lng DOUBLE NOT NULL,
	time DATETIME DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE Drone_information
{
	_id INT UNIQUE KEY AUTO_INCREMENT,
	num_drone INT PRIMARY KEY,
	
}

CREATE TABLE Client_location
(
	_id INT PRIMARY KEY AUTO_INCREMENT,
	client_id INT NOT NULL,
	lat DOUBLE NOT NULL,
	lng DOUBLE NOT NULL,
	time DATETIME DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE Client_information
(
	_id INT UNIQUE KEY AUTO_INCREMENT,
	client_id VARCHAR(20) PRIMARY KEY NOT NULL,
	password CHAR(255) NOT NULL,
	email VARCHAR(255) UNIQUE KEY,
	phone VARCHAR(20),
	time DATETIME DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE Client_logs
{
	_id INT PRIMARY KEY NOT NULL,
	client_id VARCHAR(20) NOT NULL,
	num_drone INT NOT NULL,
	time_start DATETIME DEFAULT CURRENT_TIMESTAMP,
	time_end DATETIME,
}
